Main Tables
Users
* user_id: Primary Key
* first_name: VARCHAR
* last_name: VARCHAR
* birthdate: DATE
* age: INT (calculated if needed)
* gender: ENUM('Male', 'Female', 'Non-Binary', 'Prefer not to say', 'Transgender', 'Genderqueer')
* sexual_orientation: ENUM('Heterosexual', 'Homosexual', 'Bisexual', 'Asexual', 'Pansexual', 'Queer')
* location: VARCHAR
* phone_number: VARCHAR
* email_address: VARCHAR
* bio: TEXT
Profile_Pictures
* picture_id: Primary Key
* user_id: Foreign Key References Users(user_id)
* picture_url: VARCHAR
* is_primary: BOOLEAN
